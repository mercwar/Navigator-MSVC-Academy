Chapter 33 — Direct3D Shadow Mapping

Topic Description
What is Shadow Mapping? Shadow mapping is a GPU technique that renders shadows by:
  • Rendering the scene from the light’s point of view
  • Storing depth into a shadow texture
  • Sampling that texture during normal rendering
  • Comparing fragment depth vs. light depth
Shadow mapping is used in:
  • Games
  • 3D visualization
  • Real‑time lighting
Light‑Space Matrix Convert world coordinates into light space: 

XMMATRIX lightView = XMMatrixLookAtLH(
    XMVectorSet(5, 10, -5, 1),
    XMVectorSet(0, 0, 0, 1),
    XMVectorSet(0, 1, 0, 0)
);

XMMATRIX lightProj = XMMatrixOrthographicLH(
    20.0f, 20.0f,
    1.0f, 100.0f
);

XMMATRIX lightSpace = XMMatrixMultiply(lightView, lightProj);
Orthographic projection is used for directional lights.
Shadow Depth Texture Create depth texture: 

D3D11_TEXTURE2D_DESC td = {0};
td.Width = 2048;
td.Height = 2048;
td.MipLevels = 1;
td.ArraySize = 1;
td.Format = DXGI_FORMAT_R32_TYPELESS;
td.SampleDesc.Count = 1;
td.BindFlags = D3D11_BIND_DEPTH_STENCIL |
               D3D11_BIND_SHADER_RESOURCE;

ID3D11Texture2D *shadowTex;
device->CreateTexture2D(&td, NULL, &shadowTex);
Create depth view: 

D3D11_DEPTH_STENCIL_VIEW_DESC dsvd = {0};
dsvd.Format = DXGI_FORMAT_D32_FLOAT;
dsvd.ViewDimension = D3D11_DSV_DIMENSION_TEXTURE2D;

ID3D11DepthStencilView *shadowDSV;
device->CreateDepthStencilView(shadowTex, &dsvd, &shadowDSV);
Create shader resource view: 

D3D11_SHADER_RESOURCE_VIEW_DESC srvd = {0};
srvd.Format = DXGI_FORMAT_R32_FLOAT;
srvd.ViewDimension = D3D11_SRV_DIMENSION_TEXTURE2D;
srvd.Texture2D.MipLevels = 1;

ID3D11ShaderResourceView *shadowSRV;
device->CreateShaderResourceView(shadowTex, &srvd, &shadowSRV);
Shadow Pass (Render Depth) Render scene from light’s POV: 

// Set shadow DSV
context->OMSetRenderTargets(0, NULL, shadowDSV);

// Clear depth
context->ClearDepthStencilView(shadowDSV,
    D3D11_CLEAR_DEPTH, 1.0f, 0);

// Use shadow vertex shader
context->VSSetShader(shadowVS, NULL, 0);

// No pixel shader needed
context->PSSetShader(NULL, NULL, 0);

// Draw scene geometry
context->DrawIndexed(indexCount, 0, 0);
The shadow pass writes depth only.
Shadow Vertex Shader 

cbuffer LightBuffer : register(b0)
{
    matrix world;
    matrix lightView;
    matrix lightProj;
};

struct VS_IN {
    float3 pos : POSITION;
};

struct VS_OUT {
    float4 pos : SV_POSITION;
};

VS_OUT main(VS_IN input)
{
    VS_OUT o;
    float4 p = float4(input.pos, 1);
    p = mul(p, world);
    p = mul(p, lightView);
    p = mul(p, lightProj);
    o.pos = p;
    return o;
}

// Set normal render target
context->OMSetRenderTargets(1, &rtv, depthView);

Shadow Sampling Convert world position to shadow texture coordinates: 

float4 lp = mul(float4(worldPos,1), lightView);
lp = mul(lp, lightProj);

// Convert to texture space
float2 uv = lp.xy * 0.5f + 0.5f;
float depth = lp.z;
Sample shadow map: 

float shadowDepth = shadowTex.Sample(samp, uv).r;

float shadow = depth > shadowDepth + 0.001f ? 0.3f : 1.0f;
shadow = 0.3 means in shadow. shadow = 1.0 means lit.
Pixel Shader (Lighting + Shadow) 

float4 main(float3 normal : NORMAL,
            float3 worldPos : TEXCOORD1,
            float2 uv : TEXCOORD0,
            float4 lp : TEXCOORD2) : SV_TARGET
{
    float3 N = normalize(normal);
    float3 L = normalize(-lightDir);

    float diff = saturate(dot(N, L));

    float3 V = normalize(cameraPos - worldPos);
    float3 R = reflect(-L, N);
    float spec = pow(saturate(dot(R, V)), 32);

    // Shadow mapping
    float2 suv = lp.xy * 0.5f + 0.5f;
    float depth = lp.z;
    float shadowDepth = shadowTex.Sample(sampShadow, suv).r;

    float shadow = depth > shadowDepth + 0.001f ? 0.3f : 1.0f;

    float3 texColor = tex.Sample(samp, uv).rgb;

    float3 final =
        ambientColor.rgb +
        shadow * diff * diffuseColor.rgb * texColor +
        shadow * spec * specColor.rgb;

    return float4(final, 1);
}
Full Shadow Mapping Pipeline 

// 1. Render depth from light POV → shadow map
RenderShadowPass();

// 2. Render scene normally using shadow map
RenderMainPass();

// 3. Present frame
swap->Present(1,0);
