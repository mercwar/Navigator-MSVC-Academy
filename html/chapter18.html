Chapter 18 — Icons, Accelerators & Resource‑Driven UI

Component Description
Application Icons WinAPI applications use icons defined in resource files. Resource (.rc): 

IDI_MAIN ICON "app.ico"
Load icon in window class: 

wc.hIcon = LoadIcon(hInstance, MAKEINTRESOURCE(IDI_MAIN));
wc.hIconSm = LoadIcon(hInstance, MAKEINTRESOURCE(IDI_MAIN));
Icons appear in:
  • Title bar
  • Taskbar
  • Alt‑Tab switcher
Cursors Built‑in cursors: 

wc.hCursor = LoadCursor(NULL, IDC_ARROW);
Custom cursor (.rc): 

IDC_CUSTOM CURSOR "cursor.cur"
Load: 

HCURSOR cur = LoadCursor(hInstance, MAKEINTRESOURCE(IDC_CUSTOM));
Menus from Resources Resource (.rc): 

IDR_MAINMENU MENU
BEGIN
    POPUP "&File"
    BEGIN
        MENUITEM "&Open",  1001
        MENUITEM "&Save",  1002
        MENUITEM SEPARATOR
        MENUITEM "E&xit",  1003
    END
END
Load menu: 

HMENU hMenu = LoadMenu(hInstance, MAKEINTRESOURCE(IDR_MAINMENU));
SetMenu(hwnd, hMenu);
Commands arrive in WM_COMMAND.
Accelerators (Keyboard Shortcuts) Resource (.rc): 

IDR_ACCEL ACCELERATORS
BEGIN
    "O", 1001, VIRTKEY, CONTROL
    "S", 1002, VIRTKEY, CONTROL
    "Q", 1003, VIRTKEY, CONTROL
END
Load: 

HACCEL hAccel = LoadAccelerators(hInstance, MAKEINTRESOURCE(IDR_ACCEL));
Use in message loop: 

while (GetMessage(&msg, NULL, 0, 0)) {
    if (!TranslateAccelerator(hwnd, hAccel, &msg)) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }
}
Now:
  • Ctrl+O → Open
  • Ctrl+S → Save
  • Ctrl+Q → Exit
String Tables Resource (.rc): 

STRINGTABLE
BEGIN
    IDS_HELLO   "Hello from resources!"
    IDS_ERROR   "An error has occurred."
END
Load: 

char buffer[256];
LoadString(hInstance, IDS_HELLO, buffer, 256);
MessageBox(hwnd, buffer, "Message", MB_OK);
Bitmaps Resource (.rc): 

IDB_LOGO BITMAP "logo.bmp"
Load: 

HBITMAP bmp = LoadBitmap(hInstance, MAKEINTRESOURCE(IDB_LOGO));
Draw: 

HDC mem = CreateCompatibleDC(hdc);
SelectObject(mem, bmp);
BitBlt(hdc, 10, 10, width, height, mem, 0, 0, SRCCOPY);
DeleteDC(mem);
Full Resource‑Driven UI Example 

#include <windows.h>
#include "resource.h"

LRESULT CALLBACK WndProc(HWND hwnd, UINT msg,
                         WPARAM wParam, LPARAM lParam)
{
    switch (msg) {

        case WM_COMMAND:
            switch (LOWORD(wParam)) {
                case 1001:
                    MessageBox(hwnd, "Open clicked", "Menu", MB_OK);
                    break;
                case 1002:
                    MessageBox(hwnd, "Save clicked", "Menu", MB_OK);
                    break;
                case 1003:
                    PostQuitMessage(0);
                    break;
            }
            return 0;

        case WM_PAINT: {
            PAINTSTRUCT ps;
            HDC hdc = BeginPaint(hwnd, &ps);

            char text[256];
            LoadString(GetModuleHandle(NULL), IDS_HELLO, text, 256);
            TextOut(hdc, 10, 10, text, lstrlen(text));

            EndPaint(hwnd, &ps);
            return 0;
        }

        case WM_DESTROY:
            PostQuitMessage(0);
            return 0;
    }

    return DefWindowProc(hwnd, msg, wParam, lParam);
}

int WINAPI WinMain(HINSTANCE hInstance,
                   HINSTANCE hPrevInstance,
                   LPSTR lpCmdLine,
                   int nCmdShow)
{
    WNDCLASSEX wc = {0};
    wc.cbSize        = sizeof(WNDCLASSEX);
    wc.lpfnWndProc   = WndProc;
    wc.hInstance     = hInstance;
    wc.hIcon         = LoadIcon(hInstance, MAKEINTRESOURCE(IDI_MAIN));
    wc.hIconSm       = LoadIcon(hInstance, MAKEINTRESOURCE(IDI_MAIN));
    wc.hCursor       = LoadCursor(NULL, IDC_ARROW);
    wc.hbrBackground = (HBRUSH)(COLOR_WINDOW+1);
    wc.lpszClassName = "ResourceWindow";

    RegisterClassEx(&wc);

    HWND hwnd = CreateWindowEx(
        0,
        "ResourceWindow",
        "Resource‑Driven WinAPI UI",
        WS_OVERLAPPEDWINDOW,
        CW_USEDEFAULT, CW_USEDEFAULT,
        800, 600,
        NULL, NULL,
        hInstance,
        NULL
    );

    HMENU hMenu = LoadMenu(hInstance, MAKEINTRESOURCE(IDR_MAINMENU));
    SetMenu(hwnd, hMenu);

    HACCEL hAccel = LoadAccelerators(hInstance, MAKEINTRESOURCE(IDR_ACCEL));

    ShowWindow(hwnd, nCmdShow);
    UpdateWindow(hwnd);

    MSG msg;
    while (GetMessage(&msg, NULL, 0, 0)) {
        if (!TranslateAccelerator(hwnd, hAccel, &msg)) {
            TranslateMessage(&msg);
            DispatchMessage(&msg);
        }
    }

    return 0;
}
