Chapter 16 — Menus, Buttons, and Controls in WinAPI

Component Description
Overview of Controls WinAPI provides built‑in GUI controls:
  • Menus
  • Buttons
  • Edit boxes
  • Static text
  • List boxes
  • Combo boxes
  • Check boxes
  • Radio buttons
Controls are created using CreateWindowEx with special class names.
Creating a Menu 

HMENU hMenu = CreateMenu();

AppendMenu(hMenu, MF_STRING, 1, "Open");
AppendMenu(hMenu, MF_STRING, 2, "Save");
AppendMenu(hMenu, MF_STRING, 3, "Exit");

SetMenu(hwnd, hMenu);
Menu commands are received in WM_COMMAND.
Handling Menu Commands 

case WM_COMMAND:
    switch (LOWORD(wParam)) {
        case 1:
            MessageBox(hwnd, "Open clicked", "Menu", MB_OK);
            break;
        case 2:
            MessageBox(hwnd, "Save clicked", "Menu", MB_OK);
            break;
        case 3:
            PostQuitMessage(0);
            break;
    }
    return 0;
LOWORD(wParam) contains the menu ID.
Creating a Button 

HWND btn = CreateWindowEx(
    0,
    "BUTTON",
    "Click Me",
    WS_CHILD | WS_VISIBLE,
    20, 20, 120, 40,
    hwnd,
    (HMENU)100,
    hInstance,
    NULL
);
Buttons send WM_COMMAND when clicked.
Handling Button Click 

case WM_COMMAND:
    if (LOWORD(wParam) == 100) {
        MessageBox(hwnd, "Button clicked!", "Info", MB_OK);
    }
    return 0;

Full Example Program 

#include <windows.h>

LRESULT CALLBACK WndProc(HWND hwnd, UINT msg,
                         WPARAM wParam, LPARAM lParam)
{
    switch (msg) {

        case WM_CREATE:
            CreateWindowEx(0, "BUTTON", "Click Me",
                WS_CHILD | WS_VISIBLE,
                20, 20, 120, 40,
                hwnd, (HMENU)100,
                ((LPCREATESTRUCT)lParam)->hInstance, NULL);

            CreateWindowEx(0, "STATIC", "Hello, WinAPI!",
                WS_CHILD | WS_VISIBLE,
                20, 80, 200, 25,
                hwnd, NULL,
                ((LPCREATESTRUCT)lParam)->hInstance, NULL);

            return 0;

        case WM_COMMAND:
            if (LOWORD(wParam) == 100) {
                MessageBox(hwnd, "Button clicked!", "Info", MB_OK);
            }
            return 0;

        case WM_DESTROY:
            PostQuitMessage(0);
            return 0;
    }

    return DefWindowProc(hwnd, msg, wParam, lParam);
}

int WINAPI WinMain(HINSTANCE hInstance,
                   HINSTANCE hPrevInstance,
                   LPSTR lpCmdLine,
                   int nCmdShow)
{
    WNDCLASSEX wc = {0};
    wc.cbSize        = sizeof(WNDCLASSEX);
    wc.lpfnWndProc   = WndProc;
    wc.hInstance     = hInstance;
    wc.hCursor       = LoadCursor(NULL, IDC_ARROW);
    wc.hbrBackground = (HBRUSH)(COLOR_WINDOW+1);
    wc.lpszClassName = "ControlWindow";

    RegisterClassEx(&wc);

    HWND hwnd = CreateWindowEx(
        0,
        "ControlWindow",
        "WinAPI Controls Demo",
        WS_OVERLAPPEDWINDOW,
        CW_USEDEFAULT, CW_USEDEFAULT,
        600, 400,
        NULL, NULL,
        hInstance,
        NULL
    );

    ShowWindow(hwnd, nCmdShow);
    UpdateWindow(hwnd);

    MSG msg;
    while (GetMessage(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }

    return 0;
}
