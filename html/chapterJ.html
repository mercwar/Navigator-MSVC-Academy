DXVector Consumption 

void MERCG_Init(DXVector *dx)
{
    MERCG_InitD2D(dx);
    MERCG_InitDWrite(dx);
    MERCG_InitD3D(dx);
}
Draw Dispatch 

void MERCG_Dispatch(DXVector *dx,const CY_CMD &cmd)
{
    switch(cmd.type){
        case CMD_DRAW:   MERCG_Draw(dx,cmd.payload); break;
        case CMD_TEXT:   MERCG_Text(dx,cmd.payload); break;
        case CMD_CLEAR:  MERCG_Clear(dx); break;
        case CMD_PRESENT:dx->swap->Present(1,0); break;
    }
}
