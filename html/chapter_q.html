Chapter Q — Automation

AreaRoutine
Build Automation 

File: BUILD.bat
@echo off
set SRC=C:\Projects\Navigator-MSCV
set OUT=C:\Projects\build
cl /O2 /Fe:%OUT%\app.exe %SRC%\main.cpp
echo Build complete: %OUT%\app.exe
Deployment Automation 

File: DEPLOY.bat
@echo off
set LOCAL=C:\Projects\Navigator-MSCV
set REMOTE=/var/www/html
"C:\Program Files\WinSCP\WinSCP.com" ^
  /command "open sftp://user:pass@academy-server/" ^
  "synchronize remote %LOCAL% %REMOTE%" ^
  "exit"
Packaging Automation 

File: PACKAGE.bat
@echo off
set SRC=C:\Projects\Navigator-MSCV
set OUT=C:\Projects\release.zip
"C:\Program Files\7-Zip\7z.exe" a -tzip "%OUT%" "%SRC%\*"
echo Package created: %OUT%
