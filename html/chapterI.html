DX Health Scan 

DXHealth CYHY_ScanDX(DXVector *dx)
{
    DXHealth h = {};
    h.deviceLost = MERCG_TestDevice(dx);
    h.swapValid  = dx->swap != nullptr;
    return h;
}
AVIS Health Flags 

avis.flags |= (h.deviceLost ? AVIS_FLAG_DX_REBUILD : 0);
Subsystem Pausing If DX is unhealthy: 

CYBORG_Pause();
CBORD_RebuildDX();

avis.spine.cyhy->dxHealth = h;
