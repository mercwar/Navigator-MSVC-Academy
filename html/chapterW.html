Command Routing 

void CYBORG_Route(const CY_CMD &cmd)
{
    DXVector *dx = avis.fusion.dx;

    switch(cmd.pipe){
        case PIPE_D2D:   MERCG_D2DDispatch(dx,cmd); break;
        case PIPE_DWRITE:MERCG_DWriteDispatch(dx,cmd); break;
        case PIPE_D3D:   MERCG_D3DDispatch(dx,cmd); break;
    }
}

  • “draw triangle” → D3D11
