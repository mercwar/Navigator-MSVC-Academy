Chapter 28 — Multimedia APIs: Audio, MIDI & Video

Topic Description
MMSystem Overview Windows multimedia APIs (MMSystem) provide:
  • Audio playback (waveOut)
  • Audio recording (waveIn)
  • MIDI output
  • MCI (Media Control Interface) for video
  • Timers with millisecond precision
Header: 

#include <mmsystem.h>
#pragma comment(lib, "winmm.lib")
Playing Audio with waveOut Step 1 — Define audio format: 

WAVEFORMATEX wf = {0};
wf.wFormatTag = WAVE_FORMAT_PCM;
wf.nChannels = 1;
wf.nSamplesPerSec = 44100;
wf.wBitsPerSample = 16;
wf.nBlockAlign = wf.nChannels * wf.wBitsPerSample / 8;
wf.nAvgBytesPerSec = wf.nSamplesPerSec * wf.nBlockAlign;
Step 2 — Open device: 

HWAVEOUT hWave;
waveOutOpen(&hWave, WAVE_MAPPER, &wf, 0, 0, CALLBACK_NULL);
Step 3 — Prepare buffer: 

short samples[44100]; // 1 second mono
// fill samples with audio data

WAVEHDR hdr = {0};
hdr.lpData = (LPSTR)samples;
hdr.dwBufferLength = sizeof(samples);

waveOutPrepareHeader(hWave, &hdr, sizeof(hdr));
waveOutWrite(hWave, &hdr, sizeof(hdr));
Step 4 — Cleanup: 

waveOutUnprepareHeader(hWave, &hdr, sizeof(hdr));
waveOutClose(hWave);
Recording Audio with waveIn 

HWAVEIN hIn;
waveInOpen(&hIn, WAVE_MAPPER, &wf, 0, 0, CALLBACK_NULL);

WAVEHDR hdr = {0};
hdr.lpData = (LPSTR)samples;
hdr.dwBufferLength = sizeof(samples);

waveInPrepareHeader(hIn, &hdr, sizeof(hdr));
waveInAddBuffer(hIn, &hdr, sizeof(hdr));

waveInStart(hIn);
Recording stops with: 

waveInStop(hIn);
waveInUnprepareHeader(hIn, &hdr, sizeof(hdr));
waveInClose(hIn);
MIDI Output Open MIDI device: 

HMIDIOUT hMidi;
midiOutOpen(&hMidi, 0, 0, 0, CALLBACK_NULL);
Play a note: 

// MIDI message: 0x90 = Note On, channel 0
DWORD msg = 0x007F3C90; // note 60 (middle C), velocity 127
midiOutShortMsg(hMidi, msg);
Stop note: 

DWORD off = 0x003C80; // Note Off
midiOutShortMsg(hMidi, off);
Close: 

midiOutClose(hMidi);
Playing Video with MCI MCI (Media Control Interface) plays audio/video using simple commands. Open video: 

mciSendString("open \"video.mp4\" type mpegvideo alias vid", NULL, 0, NULL);
Play: 

mciSendString("play vid", NULL, 0, NULL);
Stop: 

mciSendString("stop vid", NULL, 0, NULL);
Close: 

mciSendString("close vid", NULL, 0, NULL);
Playing Video Inside a Window 

char cmd[256];
sprintf(cmd, "play vid window %u", hwnd);
mciSendString(cmd, NULL, 0, NULL);
Resize video: 

sprintf(cmd, "put vid window at 0 0 640 480");
mciSendString(cmd, NULL, 0, NULL);
Multimedia Timers Multimedia timers offer higher precision than SetTimer. 

timeBeginPeriod(1); // 1ms resolution

UINT id = timeSetEvent(
    10, 1,        // 10ms interval
    TimerProc,    // callback
    0,
    TIME_PERIODIC
);
Callback: 

void CALLBACK TimerProc(UINT id, UINT msg,
                        DWORD user, DWORD p1, DWORD p2)
{
    // high‑precision tick
}
Stop: 

timeKillEvent(id);
timeEndPeriod(1);
Full Audio + MIDI + Video Example 

#include <windows.h>
#include <mmsystem.h>
#pragma comment(lib, "winmm.lib")

int main(void)
{
    // Play MIDI note
    HMIDIOUT hMidi;
    midiOutOpen(&hMidi, 0, 0, 0, CALLBACK_NULL);
    midiOutShortMsg(hMidi, 0x007F3C90); // Note On
    Sleep(500);
    midiOutShortMsg(hMidi, 0x003C80);   // Note Off
    midiOutClose(hMidi);

    // Play video
    mciSendString("open \"video.mp4\" type mpegvideo alias vid", NULL, 0, NULL);
    mciSendString("play vid", NULL, 0, NULL);
    Sleep(3000);
    mciSendString("close vid", NULL, 0, NULL);

    return 0;
}
