Topic Description
Constant Buffers Constant buffers store data sent to shaders every frame (matrices, lighting, etc.). Define: 

struct ConstantBuffer {
    XMMATRIX world;
    XMMATRIX view;
    XMMATRIX proj;
};
Create: 

D3D11_BUFFER_DESC bd = {0};
bd.Usage = D3D11_USAGE_DEFAULT;
bd.ByteWidth = sizeof(ConstantBuffer);
bd.BindFlags = D3D11_BIND_CONSTANT_BUFFER;

ID3D11Buffer *cb;
device->CreateBuffer(&bd, NULL, &cb);
Update: 

context->UpdateSubresource(cb, 0, NULL, &cbData, 0, 0);
context->VSSetConstantBuffers(0, 1, &cb);
Matrices (World, View, Projection) World matrix: 

XMMATRIX world = XMMatrixRotationY(angle);
View matrix: 

XMMATRIX view = XMMatrixLookAtLH(
    XMVectorSet(0, 2, -5, 1),
    XMVectorSet(0, 0, 0, 1),
    XMVectorSet(0, 1, 0, 0)
);
Projection matrix: 

XMMATRIX proj = XMMatrixPerspectiveFovLH(
    XM_PIDIV4,
    width / (float)height,
    0.1f,
    100.0f
);
Textures Load texture: 

ID3D11ShaderResourceView *srv;
D3DX11CreateShaderResourceViewFromFile(

Shaders (Textured) Vertex shader: 

cbuffer ConstantBuffer : register(b0)
{
    matrix world;
    matrix view;
    matrix proj;
};

struct VS_IN {
    float3 pos : POSITION;
    float2 uv  : TEXCOORD;
};

struct VS_OUT {
    float4 pos : SV_POSITION;
    float2 uv  : TEXCOORD;
};

VS_OUT main(VS_IN input)
{
    VS_OUT o;
    float4 p = float4(input.pos, 1.0f);
    o.pos = mul(p, world);
    o.pos = mul(o.pos, view);
    o.pos = mul(o.pos, proj);
    o.uv = input.uv;
    return o;
}
Pixel shader: 

Texture2D tex : register(t0);
SamplerState samp : register(s0);

float4 main(float2 uv : TEXCOORD) : SV_TARGET
{
    return tex.Sample(samp, uv);
}

Full Rotating Textured Cube Example 

// Full Direct3D 11 rotating cube pipeline (summary)
// Includes device, swap chain, shaders, buffers, matrices, textures.

float angle = 0.0f;

while (running) {
    angle += 0.01f;

    cbData.world = XMMatrixRotationY(angle);
    cbData.view  = view;
    cbData.proj  = proj;

    context->UpdateSubresource(cb, 0, NULL, &cbData, 0, 0);

    float clear[4] = {0.1f,0.1f,0.3f,1};
    context->ClearRenderTargetView(rtv, clear);

    context->DrawIndexed(36, 0, 0);

    swap->Present(1,0);
}
