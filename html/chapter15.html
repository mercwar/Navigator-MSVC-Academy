Chapter 15 — Drawing with GDI (Graphics Device Interface)

Topic Description
What is GDI? GDI (Graphics Device Interface) is the core Windows subsystem for drawing:
  • Text
  • Lines
  • Rectangles
  • Ellipses
  • Bitmaps
  • Custom shapes
GDI is immediate-mode: drawing happens during WM_PAINT.
WM_PAINT All drawing must occur inside a WM_PAINT handler: 

case WM_PAINT: {
    PAINTSTRUCT ps;
    HDC hdc = BeginPaint(hwnd, &ps);

    // drawing here

    EndPaint(hwnd, &ps);
    return 0;
}
HDC is the device context — the drawing surface.
Drawing text 

TextOut(hdc, 10, 10, "Hello GDI!", 11);
TextOut draws simple text at a given position.
Drawing lines 

MoveToEx(hdc, 50, 50, NULL);
LineTo(hdc, 200, 200);
MoveToEx sets the starting point. LineTo draws the line.
Drawing rectangles 

Rectangle(hdc, 20, 20, 200, 120);
Draws a rectangle using the current pen and brush.
Drawing ellipses 

Ellipse(hdc, 100, 100, 300, 250);
Draws an ellipse inside the bounding rectangle.
Pens Pens define line color and thickness: 

HPEN pen = CreatePen(PS_SOLID, 3, RGB(255,0,0));
HPEN old = SelectObject(hdc, pen);

MoveToEx(hdc, 10, 10, NULL);
LineTo(hdc, 200, 10);

SelectObject(hdc, old);
DeleteObject(pen);
Always restore and delete GDI objects.
Brushes Brushes fill shapes: 

HBRUSH brush = CreateSolidBrush(RGB(0,255,0));
HBRUSH old = SelectObject(hdc, brush);

Rectangle(hdc, 50, 50, 150, 150);

SelectObject(hdc, old);
DeleteObject(brush);
Fonts 

HFONT font = CreateFont(
    32, 0, 0, 0, FW_BOLD,
    FALSE, FALSE, FALSE,
    ANSI_CHARSET,
    OUT_DEFAULT_PRECIS,
    CLIP_DEFAULT_PRECIS,
    DEFAULT_QUALITY,
    DEFAULT_PITCH,
    "Arial"
);

HFONT old = SelectObject(hdc, font);
TextOut(hdc, 10, 50, "Big Bold Text", 13);
SelectObject(hdc, old);
DeleteObject(font);
Colors GDI uses RGB(): 

COLORREF red = RGB(255,0,0);
COLORREF green = RGB(0,255,0);
COLORREF blue = RGB(0,0,255);
Double buffering Prevent flicker by drawing to a memory DC: 

HDC mem = CreateCompatibleDC(hdc);
HBITMAP bmp = CreateCompatibleBitmap(hdc, width, height);
SelectObject(mem, bmp);

// draw to mem instead of hdc

BitBlt(hdc, 0, 0, width, height, mem, 0, 0, SRCCOPY);

DeleteObject(bmp);
DeleteDC(mem);
Full GDI Example 

#include <windows.h>

LRESULT CALLBACK WndProc(HWND hwnd, UINT msg,
                         WPARAM wParam, LPARAM lParam)
{
    switch (msg) {

        case WM_PAINT: {
            PAINTSTRUCT ps;
            HDC hdc = BeginPaint(hwnd, &ps);

            // Text
            TextOut(hdc, 10, 10, "Hello GDI!", 11);

            // Line
            MoveToEx(hdc, 20, 40, NULL);
            LineTo(hdc, 200, 40);

            // Rectangle
            Rectangle(hdc, 20, 60, 200, 140);

            // Ellipse
            Ellipse(hdc, 220, 60, 380, 140);

            EndPaint(hwnd, &ps);
            return 0;
        }

        case WM_DESTROY:
            PostQuitMessage(0);
            return 0;
    }

    return DefWindowProc(hwnd, msg, wParam, lParam);
}

int WINAPI WinMain(HINSTANCE hInstance,
                   HINSTANCE hPrevInstance,
                   LPSTR lpCmdLine,
                   int nCmdShow)
{
    WNDCLASSEX wc = {0};
    wc.cbSize        = sizeof(WNDCLASSEX);
    wc.lpfnWndProc   = WndProc;
    wc.hInstance     = hInstance;
    wc.hCursor       = LoadCursor(NULL, IDC_ARROW);
    wc.hbrBackground = (HBRUSH)(COLOR_WINDOW+1);
    wc.lpszClassName = "GDIWindow";

    RegisterClassEx(&wc);

    HWND hwnd = CreateWindowEx(
        0,
        "GDIWindow",
        "GDI Drawing Demo",
        WS_OVERLAPPEDWINDOW,
        CW_USEDEFAULT, CW_USEDEFAULT,
        800, 600,
        NULL, NULL,
        hInstance,
        NULL
    );

    ShowWindow(hwnd, nCmdShow);
    UpdateWindow(hwnd);

    MSG msg;
    while (GetMessage(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }

    return 0;
}
