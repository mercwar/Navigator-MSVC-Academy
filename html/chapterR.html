Batch Queue CYBORG batches commands before sending to MERC‑G: 

vector gpuBatch;
Batch Push 

void CYBORG_Push(const CY_CMD &cmd)
{
    gpuBatch.push_back(cmd);
}
Batch Flush 

void CYBORG_Flush()
{
    DXVector *dx = avis.fusion.dx;
    for (auto &cmd : gpuBatch)
        MERCG_Dispatch(dx,cmd);
    gpuBatch.clear();
}
