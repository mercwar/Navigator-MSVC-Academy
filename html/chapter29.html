Chapter 29 — DirectX Fundamentals: Direct2D, DirectWrite & GPU Rendering

Topic Description
Why DirectX? DirectX provides **hardware‑accelerated rendering** using the GPU. Direct2D + DirectWrite offer:
  • Fast 2D graphics
  • High‑quality text rendering
  • Anti‑aliasing
  • Transforms & effects
  • Full GPU acceleration
DirectX replaces slow GDI for modern applications.
Initializing Direct2D Include: 

#include <d2d1.h>
#pragma comment(lib, "d2d1.lib")
Create factory: 

ID2D1Factory *factory = NULL;

D2D1CreateFactory(
    D2D1_FACTORY_TYPE_SINGLE_THREADED,
    &factory

Creating a Render Target 

ID2D1HwndRenderTarget *rt = NULL;

factory->CreateHwndRenderTarget(
    D2D1::RenderTargetProperties(),
    D2D1::HwndRenderTargetProperties(hwnd,

The render target is the GPU‑accelerated drawing surface.

Full Direct2D + DirectWrite Example 

#include <windows.h>
#include <d2d1.h>
#include <dwrite.h>

#pragma comment(lib, "d2d1.lib")
#pragma comment(lib, "dwrite.lib")

ID2D1Factory *factory;
ID2D1HwndRenderTarget *rt;
ID2D1SolidColorBrush *brush;
IDWriteFactory *dwFactory;
IDWriteTextFormat *format;

void InitD2D(HWND hwnd)
{
    D2D1CreateFactory(D2D1_FACTORY_TYPE_SINGLE_THREADED, &factory);

    RECT rc;
    GetClientRect(hwnd, &rc);

    factory->CreateHwndRenderTarget(
        D2D1::RenderTargetProperties(),
        D2D1::HwndRenderTargetProperties(
            hwnd,
            D2D1::SizeU(rc.right, rc.bottom)),
        &rt
    );

    rt->CreateSolidColorBrush(
        D2D1::ColorF(D2D1::ColorF::Blue),
        &brush
    );

    DWriteCreateFactory(
        DWRITE_FACTORY_TYPE_SHARED,
        __uuidof(IDWriteFactory),
        (IUnknown**)&dwFactory
    );

    dwFactory->CreateTextFormat(
        L"Segoe UI", NULL,
        DWRITE_FONT_WEIGHT_NORMAL,
        DWRITE_FONT_STYLE_NORMAL,
        DWRITE_FONT_STRETCH_NORMAL,
        32.0f,
        L"en-us",
        &format
    );
}

LRESULT CALLBACK WndProc(HWND hwnd, UINT msg,
                         WPARAM wParam, LPARAM lParam)
{
    switch (msg) {
        case WM_CREATE:
            InitD2D(hwnd);
            return 0;

        case WM_PAINT: {
            PAINTSTRUCT ps;
            BeginPaint(hwnd, &ps);

            rt->BeginDraw();
            rt->Clear(D2D1::ColorF(D2D1::ColorF::White));

            rt->DrawLine(
                D2D1::Point2F(10, 10),
                D2D1::Point2F(200, 200),
                brush,
                3.0f
            );

            rt->DrawText(
                L"Direct2D Rendering",
                18,
                format,
                D2D1::RectF(10, 220, 400, 260),
                brush
            );

            rt->EndDraw();

            EndPaint(hwnd, &ps);
            return 0;
        }

        case WM_DESTROY:
            PostQuitMessage(0);
            return 0;
    }

    return DefWindowProc(hwnd, msg, wParam, lParam);
}

int WINAPI WinMain(HINSTANCE hInstance,
                   HINSTANCE hPrevInstance,
                   LPSTR lpCmdLine,
                   int nCmdShow)
{
    WNDCLASSEX wc = {0};
    wc.cbSize = sizeof(wc);
    wc.lpfnWndProc = WndProc;
    wc.hInstance = hInstance;
    wc.hCursor = LoadCursor(NULL, IDC_ARROW);
    wc.hbrBackground = (HBRUSH)(COLOR_WINDOW+1);
    wc.lpszClassName = "DXWindow";
    RegisterClassEx(&wc);

    HWND hwnd = CreateWindowEx(
        0,
        "DXWindow",
        "Direct2D + DirectWrite Demo",
        WS_OVERLAPPEDWINDOW,
        CW_USEDEFAULT, CW_USEDEFAULT,
        800, 600,
        NULL, NULL,
        hInstance,
        NULL
    );

    ShowWindow(hwnd, nCmdShow);
    UpdateWindow(hwnd);

    MSG msg;
    while (GetMessage(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }

    return 0;
}
