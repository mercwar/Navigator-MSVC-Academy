Chapter 27 — Owner‑Draw Controls & Advanced UI Rendering

Topic Description
What is Owner‑Draw? Owner‑draw controls allow the application to fully control how a UI element is rendered. Common owner‑draw targets:
  • Buttons
  • Listboxes
  • Menus
  • Combo boxes
  • Custom widgets
Owner‑draw is essential for:
  • Custom themes
  • Skinned interfaces
  • Advanced UI components
Owner‑Draw Button Create an owner‑draw button: 

HWND btn = CreateWindowEx(
    0,
    "BUTTON",
    "Click",
    WS_CHILD | WS_VISIBLE | BS_OWNERDRAW,
    20, 20, 120, 40,
    hwnd,
    (HMENU)100,
    hInstance,
    NULL
);
Handle drawing in WM_DRAWITEM: 

case WM_DRAWITEM: {
    DRAWITEMSTRUCT *di = (DRAWITEMSTRUCT*)lParam;

    SetBkMode(di->hDC, TRANSPARENT);

    Rectangle(di->hDC,
              di->rcItem.left,
              di->rcItem.top,
              di->rcItem.right,
              di->rcItem.bottom);

    TextOut(di->hDC,
            di->rcItem.left + 10,
            di->rcItem.top + 10,
            "Click",
            5);

    return TRUE;
}
Custom Colors & States Owner‑draw buttons can change appearance based on state: 

if (di->itemState & ODS_SELECTED) {
    FillRect(di->hDC, &di->rcItem,
             (HBRUSH)GetStockObject(GRAY_BRUSH));
} else {
    FillRect(di->hDC, &di->rcItem,
             (HBRUSH)GetStockObject(WHITE_BRUSH));
}
States:
  • ODS_SELECTED — pressed
  • ODS_DISABLED — disabled
  • ODS_FOCUS — focused

Draw items: 

case WM_DRAWITEM: {
    DRAWITEMSTRUCT *di = (DRAWITEMSTRUCT*)lParam;

    char text[256];
    SendMessage(di->hwndItem, LB_GETTEXT, di->itemID, (LPARAM)text);

    SetBkMode(di->hDC, TRANSPARENT);

    if (di->itemState & ODS_SELECTED)
        FillRect(di->hDC, &di->rcItem, (HBRUSH)GetStockObject(GRAY_BRUSH));
    else
        FillRect(di->hDC, &di->rcItem, (HBRUSH)GetStockObject(WHITE_BRUSH));

    TextOut(di->hDC, di->rcItem.left + 5, di->rcItem.top + 2,
            text, strlen(text));

    return TRUE;
}
Owner‑Draw Menus Mark menu items as owner‑draw: 

MENUITEMINFO mi = {0};
mi.cbSize = sizeof(mi);
mi.fMask = MIIM_FTYPE | MIIM_ID | MIIM_STRING;
mi.fType = MFT_OWNERDRAW;
mi.wID = 300;
mi.dwTypeData = "Custom Menu";

InsertMenuItem(hMenu, 0, TRUE, &mi);
Draw in WM_MEASUREITEM and WM_DRAWITEM.

Double‑Buffered Custom Rendering Prevent flicker: 

HDC hdc = BeginPaint(hwnd, &ps);
HDC mem = CreateCompatibleDC(hdc);

HBITMAP bmp = CreateCompatibleBitmap(hdc, width, height);
SelectObject(mem, bmp);

// draw to mem
Rectangle(mem, 0, 0, width, height);
TextOut(mem, 10, 10, "Double Buffer", 13);

// blit to screen
BitBlt(hdc, 0, 0, width, height, mem, 0, 0, SRCCOPY);

DeleteObject(bmp);
DeleteDC(mem);
EndPaint(hwnd, &ps);
Full Owner‑Draw + Custom Control Example 

#include <windows.h>
#include <string.h>

LRESULT CALLBACK CustomProc(HWND hwnd, UINT msg,
                            WPARAM wParam, LPARAM lParam)
{
    switch (msg) {
        case WM_PAINT: {
            PAINTSTRUCT ps;
            HDC hdc = BeginPaint(hwnd, &ps);

            Ellipse(hdc, 10, 10, 190, 190);
            TextOut(hdc, 60, 90, "Widget", 6);

            EndPaint(hwnd, &ps);
            return 0;
        }
    }
    return DefWindowProc(hwnd, msg, wParam, lParam);
}

LRESULT CALLBACK MainProc(HWND hwnd, UINT msg,
                          WPARAM wParam, LPARAM lParam)
{
    switch (msg) {
        case WM_CREATE:
            CreateWindowEx(
                0, "CustomWidget", NULL,
                WS_CHILD | WS_VISIBLE,
                20, 20, 200, 200,
                hwnd, NULL,
                ((LPCREATESTRUCT)lParam)->hInstance,
                NULL
            );
            return 0;

        case WM_DRAWITEM: {
            DRAWITEMSTRUCT *di = (DRAWITEMSTRUCT*)lParam;

            FillRect(di->hDC, &di->rcItem,
                     (HBRUSH)GetStockObject(WHITE_BRUSH));

            TextOut(di->hDC,
                    di->rcItem.left + 10,
                    di->rcItem.top + 10,
                    "Owner Draw",
                    10);

            return TRUE;
        }

        case WM_DESTROY:
            PostQuitMessage(0);
            return 0;
    }
    return DefWindowProc(hwnd, msg, wParam, lParam);
}

int WINAPI WinMain(HINSTANCE hInstance,
                   HINSTANCE hPrevInstance,
                   LPSTR lpCmdLine,
                   int nCmdShow)
{
    WNDCLASSEX wc = {0};
    wc.cbSize = sizeof(wc);
    wc.lpfnWndProc = MainProc;
    wc.hInstance = hInstance;
    wc.hCursor = LoadCursor(NULL, IDC_ARROW);
    wc.hbrBackground = (HBRUSH)(COLOR_WINDOW+1);
    wc.lpszClassName = "MainWindow";
    RegisterClassEx(&wc);

    wc.lpfnWndProc = CustomProc;
    wc.lpszClassName = "CustomWidget";
    RegisterClassEx(&wc);

    HWND hwnd = CreateWindowEx(
        0,
        "MainWindow",
        "Owner‑Draw & Custom Controls",
        WS_OVERLAPPEDWINDOW,
        CW_USEDEFAULT, CW_USEDEFAULT,
        900, 600,
        NULL, NULL,
        hInstance,
        NULL
    );

    ShowWindow(hwnd, nCmdShow);
    UpdateWindow(hwnd);

    MSG msg;
    while (GetMessage(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }

    return 0;
}
