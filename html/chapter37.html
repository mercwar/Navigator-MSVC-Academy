AI runs every frame and reacts to the world.
Grid Navigation Pathfinding often uses a grid: 

struct Node {
    int x, y;
    bool walkable;
    float g, h, f;
    Node *parent;
};
g = cost from start h = heuristic f = g + h
A* Pathfinding Algorithm A* finds shortest path using heuristics. Heuristic: 

float Heuristic(Node *a, Node *b)
{
    return abs(a->x - b->x) + abs(a->y - b->y);
}
Main loop: 

while (!open.empty()) {
    Node *n = GetLowestF(open);

    if (n == goal)
        return BuildPath(n);

    MoveToClosed(n);

    for (Node *m : Neighbors(n)) {
        if (!m->walkable || InClosed(m))
            continue;

        float gNew = n->g + 1;

        if (!InOpen(m) || gNew < m->g) {
            m->g = gNew;
            m->h = Heuristic(m, goal);
            m->f = m->g + m->h;
            m->parent = n;

            AddToOpen(m);
        }
    }
}
Path Smoothing After A*, smooth path by removing unnecessary nodes: 

bool ClearLine(Node *a, Node *b)
{
    // raycast or grid check
}
      

vector Smooth(vector path)
{
    vector out;
    out.push_back(path[0]);

    int last = 0;
    for (int i = 1; i < path.size(); i++) {
        if (!ClearLine(path[last], path[i])) {
            out.push_back(path[i-1]);
            last = i-1;
        }
    }

    out.push_back(path.back());
    return out;
}
Steering Behaviors Steering creates smooth movement. Seek: 

XMFLOAT3 Seek(XMFLOAT3 pos, XMFLOAT3 target)
{
    XMFLOAT3 desired = Normalize(target - pos) * maxSpeed;
    return desired - velocity;
}
Flee: 

XMFLOAT3 Flee(XMFLOAT3 pos, XMFLOAT3 threat)
{
    XMFLOAT3 desired = Normalize(pos - threat) * maxSpeed;
    return desired - velocity;
}
Arrive: 

float dist = Length(target - pos);
float speed = maxSpeed * (dist / slowRadius);
Combining Steering Weighted sum: 

XMFLOAT3 force =
    Seek(pos, target) * 1.0f +
    Avoid(obstacles) * 2.0f +
    Align(neighbors) * 0.5f;
Clamp: 

force = Clamp(force, maxForce);
Finite‑State Machine (FSM) FSM controls AI behavior: 

enum State {
    Idle,
    Patrol,
    Chase,
    Attack
};

State current;
Update: 

switch (current) {
    case Idle:
        if (SeePlayer()) current = Chase;
        break;

    case Patrol:
        if (SeePlayer()) current = Chase;
        break;

    case Chase:
        MoveToward(player);
        if (DistanceTo(player) < 2)
            current = Attack;
        break;

    case Attack:
        DoAttack();
        if (!SeePlayer()) current = Patrol;
        break;
}
Decision Trees Decision trees evaluate conditions: 

if (Health < 20)
    if (HasMedkit)
        UseMedkit();
    else
        Flee();
else
    if (SeeEnemy)
        Attack();
    else
        Patrol();
Decision trees are simple but powerful.
Behavior Trees Behavior trees use nodes:
  • Selector
  • Sequence
  • Decorator
  • Leaf actions
Example: 

Selector(
    Sequence(
        Condition(SeeEnemy),
        Action(Attack)
    ),
    Sequence(
        Condition(LowHealth),
        Action(Flee)
    ),
    Action(Patrol)
);
NavMesh Navigation NavMesh is better than grids:
  • Triangles instead of tiles
  • Smoother paths
  • Fewer nodes
  • Better for 3D worlds
Pathfinding uses polygon adjacency.
Full AI Loop 

while (running) {
    SenseWorld();
    Think();     // FSM, BT, decision tree
    Plan();      // A*, steering
    Act();       // movement, animation
}
