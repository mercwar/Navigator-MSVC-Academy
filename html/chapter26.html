Chapter 26 — Multi‑Window Applications, Child Windows & Subclassing

Topic Description
Multi‑Window Applications WinAPI allows multiple top‑level windows in the same process. Each window:
  • Has its own WNDCLASSEX
  • Has its own WndProc
  • Receives its own messages
Creating a second window: 

HWND child = CreateWindowEx(
    0,
    "ChildWindowClass",
    "Child Window",
    WS_OVERLAPPEDWINDOW,
    200, 200, 400, 300,
    NULL, NULL,
    hInstance,
    NULL
);
Show it: 

ShowWindow(child, SW_SHOW);
Child Windows Child windows are embedded inside a parent window. They use WS_CHILD and require a parent HWND. Example: 

HWND panel = CreateWindowEx(
    0,
    "STATIC",
    "",
    WS_CHILD | WS_VISIBLE | WS_BORDER,
    10, 10, 300, 200,
    hwnd,        // parent
    (HMENU)500,
    hInstance,
    NULL
);
Child windows move when the parent moves.
Custom Child Window Class You can create custom child windows with their own WndProc. Register: 

WNDCLASSEX wc = {0};
wc.cbSize = sizeof(wc);
wc.lpfnWndProc = ChildProc;
wc.hInstance = hInstance;
wc.lpszClassName = "ChildPanel";

RegisterClassEx(&wc);
Create: 

HWND panel = CreateWindowEx(
    0,
    "ChildPanel",
    NULL,
    WS_CHILD | WS_VISIBLE,
    10, 10, 300, 200,
    hwnd,
    NULL,
    hInstance,
    NULL
);
Window Subclassing Subclassing replaces a window’s WndProc with your own, allowing custom behavior. 

SetWindowLongPtr(
    hwndControl,
    GWLP_WNDPROC,
    (LONG_PTR)MySubclassProc
);
Subclass procedure: 

LRESULT CALLBACK MySubclassProc(HWND hwnd, UINT msg,
                                WPARAM wParam, LPARAM lParam)
{
    switch (msg) {
        case WM_LBUTTONDOWN:
            MessageBox(hwnd, "Clicked!", "Subclass", MB_OK);
            return 0;
    }

    return CallWindowProc(oldProc, hwnd, msg, wParam, lParam);
}
Subclassing is essential for custom controls.
Window Superclassing Superclassing creates a new window class based on an existing one. Steps:
  • Get original class info
  • Modify lpfnWndProc
  • Register new class

WNDCLASSEX wc;
GetClassInfoEx(NULL, "BUTTON", &wc);

wc.lpszClassName = "MyButton";
wc.lpfnWndProc = MyButtonProc;

RegisterClassEx(&wc);
Now you have a custom button control.
Parent/Child Communication Child → Parent: 

SendMessage(GetParent(hwnd), WM_USER + 1, 0, 0);
Parent → Child: 

SendMessage(child, WM_SETTEXT, 0, (LPARAM)"Hello");
Use custom messages: 

#define WM_CHILD_NOTIFY (WM_USER + 100)
Message Loop for Multi‑Window Apps The same message loop handles all windows: 

MSG msg;
while (GetMessage(&msg, NULL, 0, 0)) {
    TranslateMessage(&msg);
    DispatchMessage(&msg);
}
Each window’s WndProc receives only its own messages.
Full Multi‑Window Example 

#include <windows.h>

LRESULT CALLBACK ChildProc(HWND hwnd, UINT msg,
                           WPARAM wParam, LPARAM lParam)
{
    switch (msg) {
        case WM_PAINT: {
            PAINTSTRUCT ps;
            HDC hdc = BeginPaint(hwnd, &ps);
            TextOut(hdc, 10, 10, "Child Window", 12);
            EndPaint(hwnd, &ps);
            return 0;
        }
    }
    return DefWindowProc(hwnd, msg, wParam, lParam);
}

LRESULT CALLBACK MainProc(HWND hwnd, UINT msg,
                          WPARAM wParam, LPARAM lParam)
{
    switch (msg) {
        case WM_CREATE:
            CreateWindowEx(
                0,
                "ChildPanel",
                NULL,
                WS_CHILD | WS_VISIBLE,
                20, 20, 300, 200,
                hwnd,
                NULL,
                ((LPCREATESTRUCT)lParam)->hInstance,
                NULL
            );
            return 0;

        case WM_DESTROY:
            PostQuitMessage(0);
            return 0;
    }
    return DefWindowProc(hwnd, msg, wParam, lParam);
}

int WINAPI WinMain(HINSTANCE hInstance,
                   HINSTANCE hPrevInstance,
                   LPSTR lpCmdLine,
                   int nCmdShow)
{
    WNDCLASSEX wc = {0};
    wc.cbSize = sizeof(wc);
    wc.lpfnWndProc = MainProc;
    wc.hInstance = hInstance;
    wc.hCursor = LoadCursor(NULL, IDC_ARROW);
    wc.hbrBackground = (HBRUSH)(COLOR_WINDOW+1);
    wc.lpszClassName = "MainWindow";
    RegisterClassEx(&wc);

    wc.lpfnWndProc = ChildProc;
    wc.lpszClassName = "ChildPanel";
    RegisterClassEx(&wc);

    HWND hwnd = CreateWindowEx(
        0,
        "MainWindow",
        "Multi‑Window Demo",
        WS_OVERLAPPEDWINDOW,
        CW_USEDEFAULT, CW_USEDEFAULT,
        800, 600,
        NULL, NULL,
        hInstance,
        NULL
    );

    ShowWindow(hwnd, nCmdShow);
    UpdateWindow(hwnd);

    MSG msg;
    while (GetMessage(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }

    return 0;
}
