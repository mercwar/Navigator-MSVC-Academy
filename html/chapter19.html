Chapter 19 — Multithreading, Timers & Async Operations in WinAPI

Component Description
What is a Thread? A thread is an independent path of execution inside a process. WinAPI threads allow:
  • Background tasks
  • Parallel computation
  • Non‑blocking UI
  • Timers and async operations
WinAPI threads are created with CreateThread.
CreateThread 

DWORD WINAPI WorkerThread(LPVOID lpParam)
{
    for (int i = 0; i < 5; i++) {
        Sleep(1000);
    }
    return 0;
}

HANDLE hThread = CreateThread(
    NULL,
    0,
    WorkerThread,
    NULL,
    0,
    NULL
);
Sleep() pauses the thread without blocking the UI.
Closing Threads 

WaitForSingleObject(hThread, INFINITE);
CloseHandle(hThread);
Always close thread handles to avoid leaks.
Critical Sections Used to protect shared data: 

CRITICAL_SECTION cs;

InitializeCriticalSection(&cs);

EnterCriticalSection(&cs);
// modify shared data
LeaveCriticalSection(&cs);

DeleteCriticalSection(&cs);
Prevents race conditions.
Timers (SetTimer) WinAPI timers send WM_TIMER messages: 

SetTimer(hwnd, 1, 1000, NULL);  // 1 second
Handle in WndProc: 

case WM_TIMER:
    if (wParam == 1) {
        // do something every second
    }
    return 0;
Timers run on the message loop — they do NOT create threads.
KillTimer 

KillTimer(hwnd, 1);
Stops the timer.
Callback Timers Timers can call a function directly: 

void CALLBACK TimerProc(HWND hwnd, UINT msg,
                        UINT id, DWORD time)
{
    // timer callback
}

SetTimer(hwnd, 2, 500, TimerProc);
Callback timers do not require WM_TIMER.
Async File I/O WinAPI supports asynchronous file operations using OVERLAPPED: 

OVERLAPPED ov = {0};
ov.Offset = 0;

ReadFile(hFile, buffer, 1024, NULL, &ov);
Completion is signaled with:
  • GetOverlappedResult
  • WaitForSingleObject
PostMessage (Async UI Updates) 

PostMessage(hwnd, WM_USER + 1, 0, 0);
Unlike SendMessage, PostMessage is asynchronous and returns immediately.
SendMessage (Blocking) 

SendMessage(hwnd, WM_SETTEXT, 0, (LPARAM)"Hello");
SendMessage blocks until the message is processed.
Safe UI Updates from Threads Threads cannot directly modify UI. Use PostMessage: 

PostMessage(hwnd, WM_USER + 10, value, 0);
Handle in WndProc: 

case WM_USER + 10:
    // update UI safely
    return 0;
Full Thread + Timer Example 

#include <windows.h>

DWORD WINAPI WorkerThread(LPVOID lpParam)
{
    for (int i = 0; i < 5; i++) {
        Sleep(1000);
        PostMessage((HWND)lpParam, WM_USER + 1, i, 0);
    }
    return 0;
}

LRESULT CALLBACK WndProc(HWND hwnd, UINT msg,
                         WPARAM wParam, LPARAM lParam)
{
    switch (msg) {

        case WM_CREATE:
            CreateThread(NULL, 0, WorkerThread, hwnd, 0, NULL);
            SetTimer(hwnd, 1, 500, NULL);
            return 0;

        case WM_TIMER:
            if (wParam == 1) {
                MessageBeep(MB_OK);
            }
            return 0;

        case WM_USER + 1:
            // thread update
            // wParam contains counter value
            return 0;

        case WM_DESTROY:
            KillTimer(hwnd, 1);
            PostQuitMessage(0);
            return 0;
    }

    return DefWindowProc(hwnd, msg, wParam, lParam);
}

int WINAPI WinMain(HINSTANCE hInstance,
                   HINSTANCE hPrevInstance,
                   LPSTR lpCmdLine,
                   int nCmdShow)
{
    WNDCLASSEX wc = {0};
    wc.cbSize        = sizeof(WNDCLASSEX);
    wc.lpfnWndProc   = WndProc;
    wc.hInstance     = hInstance;
    wc.hCursor       = LoadCursor(NULL, IDC_ARROW);
    wc.hbrBackground = (HBRUSH)(COLOR_WINDOW+1);
    wc.lpszClassName = "AsyncWindow";

    RegisterClassEx(&wc);

    HWND hwnd = CreateWindowEx(
        0,
        "AsyncWindow",
        "WinAPI Threads & Timers",
        WS_OVERLAPPEDWINDOW,
        CW_USEDEFAULT, CW_USEDEFAULT,
        800, 600,
        NULL, NULL,
        hInstance,
        NULL
    );

    ShowWindow(hwnd, nCmdShow);
    UpdateWindow(hwnd);

    MSG msg;
    while (GetMessage(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessage(&msg);
    }

    return 0;
}
