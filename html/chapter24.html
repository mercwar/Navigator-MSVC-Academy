Chapter 24 — Windows Registry Operations

Topic Description
What is the Registry? The Windows Registry is a hierarchical database storing:
  • System configuration
  • Application settings
  • User preferences
  • Driver information
Registry operations use WinAPI functions like:
  • RegOpenKeyEx
  • RegCreateKeyEx
  • RegSetValueEx
  • RegQueryValueEx
  • RegDeleteKey
Registry Root Keys Common root keys:
  • HKEY_CLASSES_ROOT
  • HKEY_CURRENT_USER
  • HKEY_LOCAL_MACHINE
  • HKEY_USERS
  • HKEY_CURRENT_CONFIG
Opening a Registry Key 

HKEY hKey;
RegOpenKeyEx(
    HKEY_CURRENT_USER,
    "Software\\MyApp",
    0,
    KEY_READ,
    &hKey
);
Creating a Registry Key 

HKEY hKey;
DWORD disp;

RegCreateKeyEx(
    HKEY_CURRENT_USER,
    "Software\\MyApp",
    0,
    NULL,
    REG_OPTION_NON_VOLATILE,
    KEY_ALL_ACCESS,
    NULL,
    &hKey,
    &disp
);
disp tells whether the key was created or opened.
Setting a Value String value: 

char text[] = "Hello Registry";

RegSetValueEx(
    hKey,
    "Message",
    0,
    REG_SZ,
    (BYTE*)text,
    strlen(text) + 1
);
DWORD value: 

DWORD num = 123;

RegSetValueEx(
    hKey,
    "Number",
    0,
    REG_DWORD,
    (BYTE*)&num,
    sizeof(num)
);
Reading a Value 

char buffer[256];
DWORD size = sizeof(buffer);

RegQueryValueEx(
    hKey,
    "Message",
    NULL,
    NULL,
    (BYTE*)buffer,
    &size
);
DWORD: 

DWORD num;
DWORD size = sizeof(num);

RegQueryValueEx(
    hKey,

    NULL,
    (BYTE*)&num,

Deleting a Value 

RegDeleteValue(hKey, "Message");
Deleting a Key 

RegDeleteKey(HKEY_CURRENT_USER, "Software\\MyApp");
Enumerating Subkeys 

char name[256];
DWORD size = sizeof(name);
DWORD index = 0;

while (RegEnumKeyEx(
    hKey,
    index,
    name,
    &size,
    NULL,
    NULL,
    NULL,
    NULL
) == ERROR_SUCCESS)
{
    printf("Subkey: %s\n", name);
    index++;
    size = sizeof(name);
}
Enumerating Values 

char name[256];
DWORD nameSize = sizeof(name);
DWORD index = 0;

while (RegEnumValue(
    hKey,
    index,
    name,
    &nameSize,
    NULL,
    NULL,
    NULL,
    NULL
) == ERROR_SUCCESS)
{
    printf("Value: %s\n", name);
    index++;
    nameSize = sizeof(name);
}
Full Registry Example 

#include <windows.h>
#include <stdio.h>

int main(void)
{
    HKEY hKey;
    DWORD disp;

    RegCreateKeyEx(
        HKEY_CURRENT_USER,
        "Software\\DemoApp",
        0, NULL,
        REG_OPTION_NON_VOLATILE,
        KEY_ALL_ACCESS,
        NULL,
        &hKey,
        &disp
    );

    char msg[] = "Registry Demo";
    RegSetValueEx(hKey, "Message", 0, REG_SZ,
                  (BYTE*)msg, strlen(msg) + 1);

    DWORD num = 42;
    RegSetValueEx(hKey, "Number", 0, REG_DWORD,
                  (BYTE*)&num, sizeof(num));

    char buffer[256];
    DWORD size = sizeof(buffer);
    RegQueryValueEx(hKey, "Message", NULL, NULL,
                    (BYTE*)buffer, &size);

    printf("Message: %s\n", buffer);

    RegCloseKey(hKey);
    return 0;
}
