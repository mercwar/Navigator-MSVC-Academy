Chapter 32 — Direct3D Lighting: Normals, Diffuse, Specular & Phong Shading

Topic Description
Lighting Overview Direct3D lighting uses:
  • Normals
  • Light direction
  • Diffuse shading
  • Specular highlights
  • Phong illumination
Lighting is computed in shaders using math on vectors.
Normals Normals define surface orientation. Vertex structure: 

struct Vertex {
    XMFLOAT3 pos;
    XMFLOAT3 normal;
    XMFLOAT2 uv;
};
Example cube normals: 

{ {-1,-1,-1}, {0,0,-1}, {0,1} },
{ { 1,-1,-1}, {0,0,-1}, {1,1} },
...
Light Structure 

struct Light {
    XMFLOAT3 direction;
    float pad;
    XMFLOAT4 diffuse;
    XMFLOAT4 ambient;
    XMFLOAT4 specular;
};
Add to constant buffer: 

struct ConstantBuffer {
    XMMATRIX world;
    XMMATRIX view;
    XMMATRIX proj;
    Light light;
    XMFLOAT3 cameraPos;
    float pad2;
};
Diffuse Lighting Diffuse = max( dot(N, L), 0 ) * lightColor 

float3 N = normalize(input.normal);
float3 L = normalize(-light.direction);

float diffuse = saturate(dot(N, L));
Specular Lighting Specular = pow( max(dot(R, V), 0), shininess ) 

float3 R = reflect(-L, N);
float3 V = normalize(cameraPos - worldPos);

float spec = pow(saturate(dot(R, V)), 32);
Phong Illumination Final color: 

float3 color =
    light.ambient.rgb +
    diffuse * light.diffuse.rgb +
    spec * light.specular.rgb;
Vertex Shader (Lighting) 

cbuffer ConstantBuffer : register(b0)
{
    matrix world;
    matrix view;
    matrix proj;
    float3 lightDir;
    float pad;
    float4 diffuseColor;
    float4 ambientColor;
    float4 specColor;
    float3 cameraPos;
    float pad2;
};

struct VS_IN {
    float3 pos    : POSITION;
    float3 normal : NORMAL;
    float2 uv     : TEXCOORD;
};

struct VS_OUT {
    float4 pos    : SV_POSITION;
    float3 normal : NORMAL;
    float3 world  : TEXCOORD1;
    float2 uv     : TEXCOORD0;
};

VS_OUT main(VS_IN input)
{
    VS_OUT o;

    float4 wp = mul(float4(input.pos,1), world);
    o.pos = mul(mul(wp, view), proj);

    o.world = wp.xyz;
    o.normal = mul(input.normal, (float3x3)world);
    o.uv = input.uv;

    return o;
}
Pixel Shader (Phong) 

Texture2D tex : register(t0);
SamplerState samp : register(s0);

float4 main(float3 normal : NORMAL,
            float3 worldPos : TEXCOORD1,
            float2 uv : TEXCOORD0) : SV_TARGET
{
    float3 N = normalize(normal);
    float3 L = normalize(-lightDir);

    float diff = saturate(dot(N, L));

    float3 V = normalize(cameraPos - worldPos);
    float3 R = reflect(-L, N);

    float spec = pow(saturate(dot(R, V)), 32);

    float3 texColor = tex.Sample(samp, uv).rgb;

    float3 final =
        ambientColor.rgb +
        diff * diffuseColor.rgb * texColor +
        spec * specColor.rgb;

    return float4(final, 1);
}
Full Rotating Lit Cube Example 

// Summary of full pipeline:
// - Constant buffer with matrices + light
// - Vertex buffer with normals
// - Phong pixel shader
// - Rotating cube with diffuse + specular

cb.light.direction = XMFLOAT3(0.0f, -1.0f, -1.0f);
cb.light.diffuse   = XMFLOAT4(1,1,1,1);
cb.light.ambient   = XMFLOAT4(0.2f,0.2f,0.2f,1);
cb.light.specular  = XMFLOAT4(1,1,1,1);

while (running) {
    angle += 0.01f;

    cb.world = XMMatrixRotationY(angle);
    cb.view  = view;
    cb.proj  = proj;

    context->UpdateSubresource(cbGPU, 0, NULL, &cb, 0, 0);

    float clear[4] = {0.05f,0.05f,0.1f,1};
    context->ClearRenderTargetView(rtv, clear);

    context->DrawIndexed(36, 0, 0);

    swap->Present(1,0);
}
